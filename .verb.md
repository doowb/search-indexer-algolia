## Usage

```js
var algolia = require('{%= name %}');

var options = {
  APPLICATION_ID: '12345',
  SECRET_KEY: 'xxxxx',
  index: 'my-index',
  collectFn: function(file, next) {
    // customize object being collected
    var obj = {
      objectID: file.key,
      key: file.key
    };
    return next(null, obj);
  }
};
app.search.indexer('algolia', algolia(options));
```
